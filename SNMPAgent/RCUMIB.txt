rcuDeviceStatsTable OBJECT-TYPE
    SYNTAX      SEQUENCE OF RcuDeviceStatsEntry
    MAX-ACCESS  not-accessible
    STATUS      current
    DESCRIPTION
        "This table merely contains a set of data which is otherwise
         useless for true network management.  It is a table which
         describes properies about a IETF Working Group, such as the
         names of the two working group chairs.

         This example table is implemented in the
         agent/mibgroup/examples/data_set.c file."
    ::= { rcuDeviceTables 1 }

rcuDeviceStatsEntry OBJECT-TYPE
    SYNTAX      RcuDeviceStatsEntry
    MAX-ACCESS  not-accessible
    STATUS      current
    DESCRIPTION
        "A row describing a given working group"
    INDEX   { rcuValue }
    ::= {rcuDeviceStatsTable 1 }

RcuDeviceStatsEntry ::= SEQUENCE {
        rcuDeviceDescription    OCTET STRING,
        rcuDeviceIpAddress      IpAddress,
        rcuDevicePort   Gauge32,
        rcuDeviceI1     Gauge32,

        rcuDeviceI3     Gauge32,
        rcuDeviceIN     Gauge32,
        rcuDeviceU12    Gauge32,
        rcuDeviceU23    Gauge32,
        rcuDeviceU31    Gauge32,
        rcuDeviceU1N    Gauge32,
        rcuDeviceU2N    Gauge32,
        rcuDeviceU3N    Gauge32,
        frequency       Gauge32,
        power           Gauge32,
        kVAr            Gauge32,
        kVA             Gauge32,
        powerFactor     Gauge32
}
